#edit jenkins agar bisa diakses port 8080 dan 49000
docker run `
  --name jenkins-blueocean `
  --detach `
  --network jenkins `
  --env DOCKER_HOST=tcp://docker:2376 `
  --env DOCKER_CERT_PATH=/certs/client `
  --env DOCKER_TLS_VERIFY=1 `
  --publish 8080:8080
  --publish 49000:8080 `
  --publish 50000:50000 `
  --volume jenkins-data:/var/jenkins_home `
  --volume jenkins-docker-certs:/certs/client:ro `
  --volume "${HOME}:/home" `
  --restart=on-failure `
  --env JAVA_OPTS="-Dhudson.plugins.git.GitSCM.ALLOW_LOCAL_CHECKOUT=true" `
  myjenkins-blueocean:2.426.2-1


#run nginx reverse proxy 9000
docker run -d `
  --name nginx-reverse-proxy `
  --network jenkins `
  --publish 9000:9000 `
  --restart=on-failure `
  mynginx-reverse-proxy


docker build -t mynginx-reverse-proxy .
